from flask import Flask, render_template_string

app = Flask(__name__)

product = {
    "name": "حاسوب محمول",
    "price": 3500,
    "category": "إلكترونيات",
    "quantity": 5,
    "description": "حاسوب محمول بمعالج i7 وذاكرة 16GB وشاشة 15.6 بوصة."
}

html_template = """
<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <title>تفاصيل المنتج</title>
</head>
<body>
    <h2>تفاصيل المنتج</h2>
    <ul>
        <li><strong>الاسم:</strong> {{ product.name }}</li>
        <li><strong>السعر:</strong> {{ product.price }}</li>
        <li><strong>الفئة:</strong> {{ product.category }}</li>
        <li><strong>الكمية:</strong> {{ product.quantity }}</li>
        <li><strong>الوصف:</strong> {{ product.description }}</li>
    </ul>
</body>
</html>
"""

@app.route("/")
def product_details():
    return render_template_string(html_template, product=product)

app.run(debug=True)
