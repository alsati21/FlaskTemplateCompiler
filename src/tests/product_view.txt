from flask import Flask, render_template_string

app = Flask(__name__)

product = {
    "name": "حاسوب محمول",
    "price": 3500,
    "category": "إلكترونيات",
    "quantity": 5
}

html_template = """
<!DOCTYPE html>
<html lang="ar" dir="rtl">
<head>
    <meta charset="UTF-8">
    <title>عرض منتج</title>
</head>
<body>
    <h2>عرض منتج</h2>
    <table border="1">
        <tr><th>الاسم</th><td>{{ product.name }}</td></tr>
        <tr><th>السعر</th><td>{{ product.price }}</td></tr>
        <tr><th>الفئة</th><td>{{ product.category }}</td></tr>
        <tr><th>الكمية</th><td>{{ product.quantity }}</td></tr>
    </table>
</body>
</html>
"""

@app.route("/")
def view_product():
    return render_template_string(html_template, product=product)

app.run(debug=True)
