from flask import Flask, render_template_string, request

app = Flask(__name__)

# منتجات مبدئية
products = [
    {"name": "Laptop", "description": "Powerful gaming laptop", "price": 1200},
    {"name": "Phone", "description": "Smartphone with great camera", "price": 800},
    {"name": "Keyboard", "description": "Mechanical keyboard", "price": 150},
]

html_template = """
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>Products</title>

    <style>
        body {
            font-family: Arial, sans-serif;
            background-color: #f4f6f8;
        }

        .container {
            width: 50%;
            margin: 40px auto;
            background: white;
            padding: 25px;
            border-radius: 8px;
            box-shadow: 0 0 10px rgba(0,0,0,0.1);
        }

        h2 {
            text-align: center;
            margin-bottom: 20px;
        }

        .product {
            background: #e9ecef;
            margin-bottom: 15px;
            padding: 15px;
            border-radius: 6px;
        }

        .product h3 {
            margin: 0;
        }

        .price {
            color: #007bff;
            font-weight: bold;
        }

        .delete-btn {
            margin-top: 10px;
            background-color: #dc3545;
            color: white;
            border: none;
            padding: 6px 12px;
            cursor: pointer;
            border-radius: 4px;
        }

        .delete-btn:hover {
            background-color: #b02a37;
        }
    </style>
</head>

<body>
    <div class="container">
        <h2>Product List</h2>

        {% if products %}
            {% for product in products %}
            <div class="product">
                <h3>{{ product.name }}</h3>
                <p>{{ product.description }}</p>
                <p class="price">Price: ${{ product.price }}</p>

                <form method="post">
                    <input type="hidden" name="delete" value="{{ product.name }}">
                    <button class="delete-btn" type="submit">Delete</button>
                </form>
            </div>
            {% endfor %}
        {% else %}
            <p style="text-align:center;">No products available</p>
        {% endif %}
    </div>
</body>
</html>
"""

@app.route("/", methods=["GET", "POST"])
def products_page():
    if request.method == "POST":
        name_to_delete = request.form.get("delete")
        products[:] = [p for p in products if p["name"] != name_to_delete]

    return render_template_string(html_template, products=products)

app.run(debug=True)
